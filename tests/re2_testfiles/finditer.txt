Simple tests for the ``finditer`` function.
===========================================

    >>> import re2 as re
    >>> import os
    >>> mpath = os.path.abspath(os.getcwd())
    >>> mpath = os.path.join(mpath, 'tests', 're2_testfiles', 'cnn_homepage.dat')

    >>> len(list(re.finditer(r'\w+', open(mpath).read())))
    14230

    >>> [m.group(1) for m in re.finditer(r'\n#hdr-editions(.*?)\n', open(mpath).read())]
    [' a { text-decoration:none; }', ' li { padding:0 10px; }', ' ul li.no-pad-left span { font-size:12px; }']

    >>> [m.group(1) for m in re.finditer(r'^#hdr-editions(.*?)$', open(mpath).read(), re.M)]
    [' a { text-decoration:none; }', ' li { padding:0 10px; }', ' ul li.no-pad-left span { font-size:12px; }']
